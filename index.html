String(i + 1).padStart(2, "0") : (i + 1); return `${base}/page-${n}.${ext}`; }); const root = document.getElementById('issue1'); if (!root) return; const img = root.querySelector('.fb-page'); const prev = root.querySelector('.prev'); const next = root.querySelector('.next'); const count = root.querySelector('.fb-count'); const total = root.querySelector('.fb-total'); // a11y live region const live = document.createElement('div'); live.setAttribute('aria-live','polite'); live.style.position='absolute'; live.style.left='-9999px'; root.appendChild(live); let i = 0; function show(n){ i = (n + pages.length) % pages.length; img.classList.add('flip'); img.src = pages[i]; img.alt = 'Issue 1 page ' + (i + 1); count.textContent = (i + 1); total.textContent = pages.length; live.textContent = `Page ${i + 1} of ${pages.length}`; setTimeout(()=>img.classList.remove('flip'), 220); } prev.onclick = () => show(i - 1); next.onclick = () => show(i + 1); document.addEventListener('keydown', e => { if (e.key === 'ArrowRight') show(i + 1); if (e.key === 'ArrowLeft') show(i - 1); }); // touch swipe let sx=0; img.addEventListener('touchstart', e => sx = e.changedTouches[0].clientX, {passive:true}); img.addEventListener('touchend', e => { const dx = e.changedTouches[0].clientX - sx; if (Math.abs(dx) > 30) { dx < 0 ? show(i + 1) : show(i - 1); } }, {passive:true}); // prefetch neighbors img.addEventListener('load', () => { [ (i + 1) % pages.length, (i - 1 + pages.length) % pages.length ].forEach(n => { const a = new Image(); a.src = pages[n]; }); }); show(0); })();
